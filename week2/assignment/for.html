		$(function(){

		var container = $('#container'),
		examWrap = container.find('.exam_wrap'),
		examResult = null,

			changeElement = function(){
				for(var i=0; i < examWrap.length; i++){
					switch(i){
						case 0:
							examResult = examWrap.eq(0).find('.exam_q');
							drawInputElement(examResult);
							break;
						case 1:
							examResult = examWrap.eq(1).find('.exam_q');
							drawInputChkElement(examResult);
							break;
						case 2:
							examResult = examWrap.eq(2).find('.exam_q');
							drawGugudan8(examResult);
							break;
						case 3:
							examResult = examWrap.eq(3).find('.exam_q');
							drawGugudanAll(examResult);
							break;
						default:
							alert('더이상 문제가 없습니당');
							break;
					}
				}
			},

			drawInputElement = function(examResult){
				var checkNum = 0;

				while(checkNum < 5){
					examResult.append('<label>' + (checkNum+1)+'. <input type="text"></label><br>');
					checkNum++;
				}
			},

			drawInputChkElement = function(examResult){
				for(var i=1; i < 6; i++){
					if(i%2 == 0)
						examResult.append('<label>' +i+ '. <input type="text"></label><br>');
					else if(i%2 == 1){
						examResult.append('<input type="checkbox" id="chk'+i+'"><label for="chk'+i+'">' + i + '번째 Checkbox</label><br>');
					}
				}
			},

			drawGugudan8 = function(examResult){
				for(var i=1; i < 10; i++){
					examResult.append('8 X ' + i + ' = ' + (8*i)+'<br>');
				}
			},

			drawGugudanAll = function (examResult) {
				var output='',
					arr = [];
				for (var i = 2; i < 10; i++) {
					for (var j=1; j < 10; j++) {
						arr.push(i+'x'+j+'='+(i*j));
					}
				}
				examResult.get(0).innerHTML += arr.join('<br>');
			};

			changeElement();
		});
